#18: Write a Python function to check whether a number is 
#perfect or not

def perfect(a):
    b=0
    for i in range(1,int(a/2)+1):
        if a%i==0:
            b+=i
    if b==a:
        return "it is perfect number"
    else:
        return "it is not prfect number"       
a=int(input("enter a number:"))
print(perfect(a))
